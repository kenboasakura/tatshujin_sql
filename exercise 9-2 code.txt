SELECT DISTINCT
	employee
FROM
	employee_skills AS es1
WHERE 
	NOT EXISTS
	(
		SELECT
			skill 
		FROM
			skills
		EXCEPT
		SELECT
			skill
		FROM
			employee_skills AS es2
		WHERE
			es1.employee = es2.employee
	)
AND
	NOT EXISTS
	(
		SELECT
			skill
		FROM
			employee_skills es3
		WHERE
			es1.employee = es3.employee
		EXCEPT
		SELECT
			skill 
		FROM
			skills
	)
;
