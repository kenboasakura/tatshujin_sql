SELECT
	MASTER.age_class,
	MASTER.sex_cd,
	SUM(
		CASE
		WHEN pref_name IN ('青森','秋田') THEN population
		ELSE NULL
		END
	)AS pop_tohoku,
	SUM(
		CASE
		WHEN pref_name IN ('東京','千葉') THEN population
		ELSE NULL
		END
	)AS pop_kanto
FROM
	(
		SELECT
			age_class,
			sex_cd
		FROM
			tblage
		CROSS JOIN
			tblsex
	)MASTER
LEFT JOIN
	tblpop AS DATA
ON
	MASTER.age_class = DATA.age_class
AND
	MASTER.sex_cd = DATA.sex_cd
GROUP BY
	MASTER.age_class,
	MASTER.sex_cd
;
