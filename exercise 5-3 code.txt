	digits
	(
		digit INTEGER PRIMARY KEY
	)
;

CREATE VIEW
	numbers (number) AS
SELECT
	/*連番を999まで作るためにかけられる整数*/
	first_digits.digit  + (second_digits.digit * 10) + (third_digits.digit * 100)
FROM
	digits AS first_digits
CROSS JOIN
	digits AS second_digits
CROSS JOIN
	digits AS third_digits
;

SELECT DISTINCT
	first_numbers.number
FROM
	numbers AS first_numbers
WHERE
	/*1より大きい番号*/
	first_numbers.number > 1
AND
	/*100以下の番号*/
	first_numbers.number <= 100 
AND
	/*0以外の整数すべて*/
	0 <> ALL(
		SELECT
			MOD(second_numbers.number,third_numbers.number)
		FROM
			numbers AS second_numbers
		CROSS JOIN
			numbers AS third_numbers
		ON
			second_numbers.number > third_numbers.number
		WHERE
			first_numbers.number = second_numbers.number
		AND
			/*1より大きい整数*/
			third_numbers.number > 1
	)
;
